// Schemas and Objects:
----------------------

    Customer Schema:
    ---------------
        customerId: String
        userName: String
        password: String
        name: String
        profilePic: image <File>
        email: String
        mobile: Number
        address: String
        pincode: Number

    Restaurant Schema:
    -----------------
        restaurantId: String
        restaurantName: String 
        restaurantAddress: String
        restaurantPincode: Number
        restaurantMobile: Number
        restaurantPhoto: image <File>
        restaurantAmbience: image <Array>
        restaurantRating: Number
        menu: Object {
            starters: Array < type: Food > ,
            mainCourse: Array < type: Food > ,
            dessert: Array < type: Food > ,
            juices: Array < type: Food >
        }

    Food Schema:
    -----------
        object : {
            restaurantId: String < restaurantId >
            foodId: String
            img: File
            name: String
            type: String < starters / mainCourse / dessert / juices >
            category: String 
            veg: Boolean < True/ False>
            foodRating: Number
            price: Number
            discount: Number
            available: Boolean < True / False >
        }

    Orders Schema:
    -------------
        orderId: String
        restaurant: String <restaurantId>
        customer: String <customerId>
        deliveryPerson: String <deliveryPersonId>
        food: Array < Object > {
            foodId: String,
            quantity: Number,
            cost: Number
        }
        deliveryCost: Number < 5% totalOrder Cost >
        totalCost: Number
        state: String < 'Pending' / 'AllotedDelivery' / 'Cooked' / 'Picked' / 'Completed' / 'Cancelled' >

    deliveryPerson Schema:
    ---------------------
        deliveryPersonId: String
        name: String
        email: String
        mobile: Number
        penalties: Array <penaltiesId>
        deliveryPersonRating: Number

    Penalties Schema: 
    ----------------
        penaltiesId: String
        order: String <orderId>
        deliveryPerson: String <deliveryPersonId>
        penalityCost: Number
        paid: Boolean < True / False >
